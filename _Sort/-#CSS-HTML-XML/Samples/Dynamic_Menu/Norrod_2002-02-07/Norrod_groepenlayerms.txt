function placeLayer(vert, mainlyr, sublyr, rollnme){
	// remove possible previous layer
	if(aLayer != ""){
		window.clearTimeout(timeID);
		hideLayer(aLayer, -500, -500);
	}

	// rollover image
	rollImg(placeLayer.arguments[1], placeLayer.arguments[3]);

	// place a new layer
	aLayer = placeLayer.arguments[2];
	if (NS){
		if(document.layers[sublyr].document.images.length==0){return;}
		theLeft = document.layers[mainlyr].left;
		theWidth = document.layers[mainlyr].clip.width;
		if(placeLayer.arguments[1] == "groepen02"){
			thePos = document.layers["groepen01"].left + 10;			// was  + 20
		}
		else{
			thePos = document.layers["groepen02"].left - 10;			// devide by 2
		}
		if(vert - 8 < 0){vert = 8}
		document.layers[sublyr].moveTo(thePos, vert - 8);
		document.layers["schaduw"].clip.height = (document.layers[sublyr].document.images.length * 15) + 2
		document.layers["schaduw"].moveTo((thePos + 4), (vert - 4));
		mainGroupImage = document.layers[mainlyr].document.images[rollnme];
	}
	else{
		if(document.all[sublyr].all.length==2){return;}
		theLeft = parseInt(document.all[mainlyr].style.left);
		theWidth = parseInt(document.all[mainlyr].style.width);
		if(placeLayer.arguments[1] == "groepen02"){
			thePos = parseInt(document.all["groepen01"].style.left) + 10;
		}
		else{
			thePos = parseInt(document.all["groepen02"].style.left) - 10;
		}
		if(vert - 8 < 0){vert = 8}
		document.all[sublyr].style.left = thePos;

		myHeight = (document.all[sublyr].all.length - 2) / 6 * 15			// talking about reusable :)
		document.all["schaduw"].style.height = myHeight + 2
		document.all["schaduw"].style.left = thePos + 4;
		document.all["schaduw"].style.top = vert - 4;
		mainGroupImage = document.all[rollnme];
	}

function selecttoview(myGroup, mySelection, layerNme, imagesId){		// (groupname && sub-layername), ("all" of integer), (mainlayername), (name of hilite image)
	if(multimode==false){
		document.bestelling.bestellijst.value = "#" + myGroup + "*" + mySelection + "*";
		document.bestelling.submit()
		return;
	}
	//if already some selected, start munching
	if(theSelection.indexOf(myGroup) != -1){
		// cut the string to pieces
		var firstPart = "";
		var midPart = "";
		var endPart = "";
		firstPart = theSelection.substring(0, (theSelection.indexOf(myGroup) + myGroup.length ));
		// if there's only one groep there's no "#" left
		if(theSelection.indexOf("#", firstPart.length) == -1){
			midPart = theSelection.substring(firstPart.length, theSelection.length);
		}
		else{
			midPart = theSelection.substring(firstPart.length, theSelection.indexOf("#", firstPart.length));
		}
		start = firstPart.length + midPart.length;
		end = theSelection.length;
		endPart = theSelection.substring(start, end);

		if(mySelection == "all"){			// maingroup-name is pressed
				if(midPart == "*all*"){
					// all deselected
					firstPart = firstPart.substring(0, firstPart.lastIndexOf("#"));
					midPart = "";
					// deselect all images in the sublayer, var aLayer contains it's name
					submenuselection(myGroup)
					indicateSelection(layerNme, imagesId, "blok_neut");
				}
				else{
					// all selected
					midPart = "*all*";
					// deselect all images in the sublayer, var aLayer contains it's name
					submenuselection(myGroup)
					indicateSelection(layerNme, imagesId, "blok_loss_000");
				}
		}
		else{						// subgroup is selected
				if(midPart == "*all*"){
					midPart = "*" + mySelection + "*";
					indicateSelection(layerNme, imagesId, "blok_loss_000");
					// group = partially selected
					mainGroupImage.src =  "images/blok_loss_dem.gif";
				}
				else{								//subgroup is selected or deselected midPart will look like *12*30*
					if(midPart.indexOf("*" + mySelection + "*") == -1){	// it's NOT selected yet
						// select subgroup
						indicateSelection(layerNme, imagesId, "blok_loss_000");
						if(midPart == ""){
							midPart = "*" + mySelection + "*";
						}
						else{
							midPart = midPart + mySelection + "*";
						}
					}
					else{							// it IS selected, remove it
						// deselect subgroup
						midPart = removeSubString(midPart, mySelection + "*");
						indicateSelection(layerNme, imagesId, "blok_neut");
						if(midPart == "*"){
							//remove entire group
							midPart = "";
							firstPart = firstPart.substring(0, firstPart.lastIndexOf("#"));
							// deselect maingroup-image
							mainGroupImage.src =  "images/blok_neut.gif";
						}
					}
				}	
		}
		theSelection = firstPart + midPart + endPart;
	}
	// if none yet selected, add to selection list
	else{
		// (partially) selected
		if(mySelection != "all"){mainGroupImage.src =  "images/blok_loss_dem.gif";}
		indicateSelection(layerNme, imagesId, "blok_loss_000");
		theSelection = theSelection + "#" + myGroup + "*" + mySelection + "*";
	}
	document.bestelling.bestellijst.value = theSelection;
}

function submenuselection(lyer){
	if(NS){
		times=document.layers[lyer].document.images.length;
		for(t=0; t<times; t++){
			document.layers[lyer].document.images[t].src="images/blok_neut.gif"
		}
	}
	else{
		times=document.all[lyer].all.length;
		for(t=0; t<times; t++){
			if(document.all[lyer].all[t].name!=null){
				if(document.all[lyer].all[t].name.indexOf("nme") !=-1){
					document.all[lyer].all[t].src="images/blok_neut.gif"
				}
			}
		}
	}
}
// reset all images in the selection menu
function wisselectie(){
	theSelection = "";
	document.bestelling.bestellijst.value = theSelection;
	if(NS){
		times=document.layers.length;
		for(n=0; n < times; n++){
			numof=document.layers[n].document.images.length;
			for(t=0; t < numof; t++){
				if(document.layers[n].document.images[t].name.indexOf("nme") !=-1){document.layers[n].document.images[t].src="images/blok_neut.gif"}
			}
		}
	}
	else{
		times=document.images.length;
		for(n=0; n < times; n++){
			if(document.images(n).name != null){
				if(document.images(n).name.indexOf("nme")!=-1){document.images(n).src="images/blok_neut.gif"}
			}
		}
	}
}
function removeSubString(str, toRemove){
	if(str.indexOf(toRemove) != -1){
		frst = str.substring(0, str.indexOf(toRemove));
		end = str.substring(frst.length + toRemove.length, str.length);
		retstr = frst + end;
		return retstr;
	}
}

function refresh(){
	if(NS){
		document.layers["menu"].moveTo(document.layers["menu"].left, calcWinSize("y") - document.layers["menu"].clip.height);
		parent.code.LS()
	}
	else{
		document.all["menu"].style.left = parseInt(document.all["menu"].style.left);
		document.all["menu"].style.top = calcWinSize("y") - parseInt(document.all["menu"].style.height);
	}
}
function calcWinSize(dim){
	if(dim=="x"){
		if(NS){
			return window.innerWidth;
		}
		else{
			return window.overzicht.document.body.clientWidth;
		}
	}
	else{
		if(NS){
			return window.innerHeight;
		}
		else{
			return top.frames["overzicht"].document.body.clientHeight;
		}
	}
}
